ERS DDL

create table ers_user_roles (
    ur_id serial primary key,
    ur_role varchar(40) 
    );
    

create table ers_users (
    u_id     serial primary key,
    u_username varchar(40) unique not null,
    u_password varchar(40) not null,
    u_firstname varchar(30) ,
    u_lastname varchar(30) ,
    u_email varchar(100) unique,
    ur_id  integer not null references ers_user_roles(ur_id)
    ) ;

create table ers_reimbursement_status (
    rs_id serial primary key,
    rs_status varchar(30) not null
    );

create table ers_reimbursement_type (
    rt_id serial primary key,
    rt_type varchar(30) not null
    );

create table ers_reimbursements (
    r_id serial primary key,
    r_amount numeric(22,2) not null,
    r_description varchar(100),
    r_receipt bytea,
    r_submitted timestamp not null,
    r_resolved timestamp,
    u_id_author integer not null references ers_users(u_id),
    u_id_resolver integer references ers_users(u_id),
    rt_type integer not null references  ers_reimbursement_type(rt_id),
    rt_status integer not null references ers_reimbursement_status(rs_id)
    );
